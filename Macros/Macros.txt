#Persistent            ; keep the script running
SetBatchLines, -1     ; run at full speed if needed

; ---- CONFIGURATION ----
Interval := 5000      ; run every 5000 ms (5 seconds). Change as needed.

; ---- STATE VARIABLE ----
running := false

; ---- TOGGLE WITH F9 ----
F9::
    running := !running
    if (running) {
        SetTimer, DoMacro, %Interval%
        TrayTip,, Macro ON (every %Interval% ms), 1
    } else {
        SetTimer, DoMacro, Off
        TrayTip,, Macro OFF, 1
    }
return

; ---- YOUR MACRO SEQUENCE ----
DoMacro:
    ; 1) Ctrl+Del
    Send, ^{Del}
    Sleep, 50
